398    def isAFN(self, cell):
399        """
400        All Free Neighbours kijkt of het aantal bommen voldaan is, dus als een cel de value heeft van 1 en er ligt 1 bom
401        omheen dan is deze conditie voldaan
402
403        :param cell: Cell
404            Welke cell er naar gekeken moet worden
405        :return:
406            of de conditie is voldaan
407        """
408        # all free neighbours
409        bombs = 0
410        for x in cell.neighbours:
411            if x.flagged:
412                bombs += 1
413
414        return True if bombs == cell.value else False

416    def isAMN(self, cell):
417        """All Marked Neighbours kijkt of het aantal dichte cellen + aantal vlaggen gelijk is aan de waarde van de cel
418
419        :param cell: Cell
420            welke cell er naar gekeken moet worden
421        :return:
422            of de conditie is voldaan
423        """
424        # all marked neighbours
425        cells = 0
426        for x in cell.neighbours:
427            if not x.revealed or x.flagged:
428                cells += 1
429
430        return True if cells == cell.value else False
